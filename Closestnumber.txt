x = 300

def closestnum():

      num1 = int(input("Enter your First number: "))
      num2 = int(input("Enter your Second number: "))
      num3 = int(input("Enter your Third number: "))
      
      closest_number = min([num1,num2,num3], key = lambda num: abs(num-x))
      
      print(f'The closest value to {x} is {closest_number}')
      
closestnum()
